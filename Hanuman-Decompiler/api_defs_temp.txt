# Abhi-Decompiler API Definitions
# Supports TWO formats - both can be mixed in this file:
#
# Format 1 - Pipe separated (compact):
#   CreateProcessA|BOOL|LPCSTR lpApplicationName|LPSTR lpCommandLine|...
#
# Format 2 - MSDN paste (paste directly from docs.microsoft.com):
#   BOOL CreateProcessA(
#     [in, optional]  LPCSTR  lpApplicationName,
#     [in, out]       LPSTR   lpCommandLine,
#     ...
#   );
#
# Lines starting with # are comments.
# Place this file next to Abhi-Decompiler.dp64 / .dp32
